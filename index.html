}) .catch(error => { console.error('Error:', error); }); } from flask import Flask, request, jsonify from PIL import Image import torch import torchvision.transforms as transforms import os # Initialize the Flask app app = Flask(__name__) # Load the trained model device = torch.device("cuda:0" if torch.cuda.is_available() else "cpu") model = torch.load('clean_polluted_river_model.pth', map_location=device) model.eval() # Image transformations test_transforms = transforms.Compose([ transforms.Resize(256), transforms.CenterCrop(224), transforms.ToTensor(), transforms.Normalize([0.485, 0.456, 0.406], [0.229, 0.224, 0.225]) ]) # Define labels dataset_labels = ["clean", "polluted"] # Route to classify the image @app.route('/classify', methods=['POST']) def classify_image(): # Get the water body name and file from the request water_body_name = request.form.get('water-body-name') file = request.files['file-upload'] if not file: return jsonify({'error': 'No file uploaded'}), 400 # Load the image image = Image.open(file).convert('RGB') # Transform the image image_tensor = test_transforms(image).unsqueeze(0).to(device) # Perform inference with torch.no_grad(): outputs = model(image_tensor) _, preds = torch.max(outputs, 1) # Get the prediction label prediction = dataset_labels[preds[0].item()] # Send the result back to the client return jsonify({ 'water_body': water_body_name, 'prediction': prediction }) if __name__ == '__main__': app.run(debug=True, port=5000)
